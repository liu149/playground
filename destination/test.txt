{{- if .Values.ingress.enabled }}
{{- $hosts := list }}
{{- if .Values.ingress.hosts }}
  {{- $hosts = .Values.ingress.hosts }}
{{- else if .Values.ingress.host }}
  {{- if kindIs "slice" .Values.ingress.host }}
    {{- $hosts = .Values.ingress.host }}
  {{- else }}
    {{- $hosts = list .Values.ingress.host }}
  {{- end }}
{{- else if .Values.dns.baseDomain }}
  {{- $hosts = list (printf "lib-store-explorer.%s" .Values.dns.baseDomain) }}
{{- end }}
{{- $servicePort := .Values.service.port | default 8080 }}
{{- $ingressPath := .Values.ingress.path | default "/" }}
{{- $ingressPathType := .Values.ingress.pathType | default "Prefix" }}
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: lib-store-explorer
  namespace: {{ .Values.namespace }}
  {{- with .Values.ingress.annotations }}
  annotations:
{{ toYaml . | indent 4 }}
  {{- end }}
spec:
  ingressClassName: {{ .Values.ingress.className | default "nginx" }}
  {{- if .Values.ingress.tls.enabled }}
  tls:
    - hosts:
{{- range $hosts }}
        - {{ . }}
{{- end }}
      secretName: {{ .Values.ingress.tls.secretName | default "tls-secret" }}
  {{- end }}
  rules:
{{- range $hosts }}
    - host: {{ . }}
      http:
        paths:
          - path: {{ $ingressPath }}
            pathType: {{ $ingressPathType }}
            backend:
              service:
                name: lib-store-explorer-service
                port:
                  number: {{ $servicePort }}
{{- end }}
{{- end }}
